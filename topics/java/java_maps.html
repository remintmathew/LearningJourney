Java Maps and IC Validation

A Map is an object that maps keys to values. A map cannot contain duplicate keys; each key can map to at most one value.

Overview of Maps

Maps are part of the Java Collections Framework and are used for storing data in key-value pairs. They provide a convenient way to associate keys with values and to retrieve values based on their keys.

Types of Maps

There are several types of maps in Java:

  • HashMap
  • LinkedHashMap
  • TreeMap
  • Hashtable

HashMap vs LinkedHashMap

HashMap: HashMap makes no guarantees as to the order of the map; in particular, it does not guarantee that the order will remain constant over time.

LinkedHashMap: LinkedHashMap is an ordered version of HashMap that maintains a doubly-linked List across all entries. This linked list defines the iteration ordering, which is normally the order in which keys were inserted into the map (insertion-order).

Adding Values to a Map

// Importing necessary classes
import java.util.HashMap;
import java.util.Map;

public class Main {
    public static void main(String[] args) {
        // Creating a HashMap
        Map map = new HashMap<>();

        // Adding values to the map
        map.put("One", 1);
        map.put("Two", 2);
        map.put("Three", 3);

        // Displaying the map
        System.out.println(map);
    }
}

Iterating over a Map

// Importing necessary classes
import java.util.HashMap;
import java.util.Map;

public class Main {
    public static void main(String[] args) {
        // Creating a HashMap
        Map map = new HashMap<>();

        // Adding values to the map
        map.put("One", 1);
        map.put("Two", 2);
        map.put("Three", 3);

        // Iterating over the map
        for (Map.Entry entry : map.entrySet()) {
            System.out.println(entry.getKey() + " = " + entry.getValue());
        }
    }
}

Maps are useful for storing and managing test results, voltages, currents, and other measurements across different sites and conditions.

Storing Voltage Values per Site

// Importing necessary classes
import java.util.HashMap;
import java.util.Map;

public class ICValidation {
    public static void main(String[] args) {
        // Creating a map to store voltage values per site
        Map voltageMap = new HashMap<>();

        // Adding voltage values
        voltageMap.put("Site1", 1.1);
        voltageMap.put("Site2", 1.2);
        voltageMap.put("Site3", 1.3);

        // Displaying the voltage values
        for (Map.Entry entry : voltageMap.entrySet()) {
            System.out.println(entry.getKey() + " = " + entry.getValue() + "V");
        }
    }
}

Checking if a Value is Present

// Importing necessary classes
import java.util.HashMap;
import java.util.Map;

public class ICValidation {
    public static void main(String[] args) {
        // Creating a map to store voltage values per site
        Map voltageMap = new HashMap<>();

        // Adding voltage values
        voltageMap.put("Site1", 1.1);
        voltageMap.put("Site2", 1.2);
        voltageMap.put("Site3", 1.3);

        // Checking if a site is present
        String siteToCheck = "Site2";
        if (voltageMap.containsKey(siteToCheck)) {
            System.out.println(siteToCheck + " is present with voltage " + voltageMap.get(siteToCheck) + "V");
        } else {
            System.out.println(siteToCheck + " is not present.");
        }
    }
}

Map of Maps for Storing Scan Results

// Importing necessary classes
import java.util.HashMap;
import java.util.Map;

public class ICValidation {
    public static void main(String[] args) {
        // Creating a map of maps to store scan results
        Map> scanResults = new HashMap<>();

        // Creating inner maps for different tests
        Map test1Results = new HashMap<>();
        test1Results.put("Site1", 100);
        test1Results.put("Site2", 95);

        Map test2Results = new HashMap<>();
        test2Results.put("Site1", 98);
        test2Results.put("Site2", 99);

        // Adding test results to the main map
        scanResults.put("Test1", test1Results);
        scanResults.put("Test2", test2Results);

        // Displaying the scan results
        for (Map.Entry> testEntry : scanResults.entrySet()) {
            System.out.println(testEntry.getKey() + " results:");
            for (Map.Entry siteEntry : testEntry.getValue().entrySet()) {
                System.out.println("  " + siteEntry.getKey() + ": " + siteEntry.getValue());
            }
        }
    }
}

Map of ArrayList for Storing Multiple Measurements

// Importing necessary classes
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class ICValidation {
    public static void main(String[] args) {
        // Creating a map of ArrayList to store multiple measurements per site
        Map> measurementsMap = new HashMap<>();

        // Adding measurements
        measurementsMap.computeIfAbsent("Site1", k -> new ArrayList<>()).add(1.1);
        measurementsMap.computeIfAbsent("Site1", k -> new ArrayList<>()).add(1.15);
        measurementsMap.computeIfAbsent("Site2", k -> new ArrayList<>()).add(1.2);

        // Displaying the measurements
        for (Map.Entry> entry : measurementsMap.entrySet()) {
            System.out.println(entry.getKey() + " measurements: " + entry.getValue());
        }
    }
}

Using computeIfAbsent

// Importing necessary classes
import java.util.HashMap;
import java.util.Map;

public class ICValidation {
    public static void main(String[] args) {
        // Creating a map to store current values per site
        Map currentMap = new HashMap<>();

        // Adding current values using computeIfAbsent
        currentMap.computeIfAbsent("Site1", k -> 0.5);
        currentMap.computeIfAbsent("Site2", k -> 0.7);

        // Displaying the current values
        for (Map.Entry entry : currentMap.entrySet()) {
            System.out.println(entry.getKey() + " = " + entry.getValue() + "A");
        }
    }
}
